    },
    "1000124567": {
      title: "轻盈防晒霜 SPF 50+ PA++++",
      image: "https://p1.tiimg.com/100/1000/1000015y1f0I2y0J0B.jpg",
      price: { nowPrice: "¥89.9", marketPrice: "¥129.00" },
      activity: "新人专享",
      discount: "30%",
      stock: "有货",
      tags: ["防晒","免费样本"]
    },
    "1000136789": {
      title: "智能蓝牙耳机",
      image: "https://p1.tiimg.com/100/1000/1000015y1f0I2y0J0C.jpg",
      price: { nowPrice: "¥199", marketPrice: "¥399" },
      activity: "限时85折",
      discount: "85%",
      stock: "预售",
      tags: ["蓝牙5.0","降噪"]
    }
    // … add more SKUs as needed …
  };

  // ---------------------- UI helpers --------------------
  const $ = id => document.getElementById(id);
  const show = el => el.classList.remove('hidden');
  const hide = el => el.classList.add('hidden');

  // ---------------------- main logic --------------------
  const searchBtn = $('searchBtn');
  const skuInput  = $('skuInput');
  const loading   = $('loading');
  const result    = $('result');

  searchBtn.addEventListener('click', runQuery);
  skuInput.addEventListener('keypress', e => {
    if (e.key === 'Enter') runQuery();
  });

  function runQuery() {
    const sku = skuInput.value.trim();
    if (!sku) return;

    // basic validation – you can make this stricter if needed
    if (!/^\d+$/.test(sku)) {
      alert('请输入数字形式的 SKU 编号');
      return;
    }

    hide(result);
    show(loading);
    // simulate async latency
    setTimeout(() => {
      const prod = catalog[sku];
      if (prod) {
        result.innerHTML = `
          <div class="product">
            <div class="thumb" style="background-image:url('${prod.image}');"></div>
            <div class="info">
              <div class="title">${prod.title}</div>
              <div class="sku">SKU: ${sku}</div>
              <div class="price">
                <span>当前价：</span> <strong>${prod.price.nowPrice}</strong>
                <span>市场价：</span> ${prod.price.marketPrice}
              </div>
              <div class="tags">
                ${prod.tags.map(t => `<span class="tag">${t}</span>`).join('')}
              </div>
              <div class="stock">库存状态：${prod.stock}</div>
            </div>
          </div>`;
      } else {
        result.innerHTML = `<p style="color:#999;">未找到对应的商品，请检查 SKU 是否正确。</p>`;
      }
      hide(loading);
      show(result);
    }, 600);
  }
</script>
